use std::sync::mpsc;
use std::thread;

fn main() {
  let (tx, rx) = mpsc::channel();

  let handle = thread::spawn(move || {
    if let Ok(msg) = rx.recv() {
      println!("Rust got: {}", msg);
    }
  });

  // Send a message
  let _ = tx.send("Hello from Rust".to_string());

  handle.join().unwrap();
}
Pid = spawn(fun() ->

end),
